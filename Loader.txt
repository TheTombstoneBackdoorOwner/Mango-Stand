local function hookOwner(plr)
    if plr.Name ~= Account then return end
    plr.Chatted:Connect(function(msg)
        msg = msg:lower()
        if msg == ".tton" then
            trashtalk = true
        elseif msg == ".ttoff" then
            trashtalk = false
        end
    end)
end

for _, plr in ipairs(Players:GetPlayers()) do
    hookOwner(plr)
end

Players.PlayerAdded:Connect(hookOwner)

task.spawn(function()
    while true do
        if trashtalk then
            pcall(function()
                TextChatService.TextChannels.RBXGeneral:SendAsync(
                    trashtalkmsgs[math.random(1, #trashtalkmsgs)]
                )
            end)
            task.wait(3)
        else
            task.wait(0.2)
        end
    end
end)
